terminal Playground: Deploy Agent

const omni = require('omni-ai');

 

// Configure your first voice agent

async function deploy() {

  const response = await omni.agents.deploy({

    model: 'omni-voice-v2',

    capabilities: ['telephony', 'scheduling'],

    memory_enabled: true

  });

  console.log(`Agent Live: ${response.id}`);

}

 

deploy();
